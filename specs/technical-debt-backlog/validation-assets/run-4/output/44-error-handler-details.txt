import { Request, Response, NextFunction } from 'express';
import { ZodError } from 'zod';
import { AppError } from '../errors';
import { captureException } from '../services/monitoring';

const isProduction = process.env.NODE_ENV === 'production';

/**
 * Map Prisma known request error codes to HTTP responses.
 */
function mapPrismaError(err: unknown): { statusCode: number; code: string; message: string } | null {
  if (
    err &&
    typeof err === 'object' &&
    'code' in err &&
    typeof (err as Record<string, unknown>).code === 'string'
  ) {
    const prismaCode = (err as Record<string, unknown>).code as string;
    switch (prismaCode) {
      case 'P2002':
        return { statusCode: 409, code: 'CONFLICT', message: 'A record with this value already exists' };
      case 'P2025':
        return { statusCode: 404, code: 'NOT_FOUND', message: 'Record not found' };
      case 'P2003':
        return { statusCode: 400, code: 'VALIDATION_ERROR', message: 'Related record not found (foreign key constraint)' };
      default:
        return null;
    }
  }
  return null;
}

/**
 * Centralized error handler middleware.
 *
 * Handles:
 * 1. AppError subclasses — uses statusCode and code from the error
 * 2. Prisma known errors — maps P2002/P2025/P2003 to appropriate HTTP codes
 * 3. ZodError — maps to 400 with flattened field errors
 * 4. Unknown errors — 500 Internal Server Error
 *
 * Sentry: only reports 5xx errors.
 */
export function errorHandler(err: Error, req: Request, res: Response, _next: NextFunction): void {
  // 1. AppError subclasses
  if (err instanceof AppError) {
    if (err.statusCode >= 500) {
      req.log.error({ err, path: req.path, method: req.method }, 'Server error');
      captureException(err, {
        requestId: req.requestId,
        path: req.path,
        method: req.method,
        userId: req.userId,
      });
    } else {
      req.log.warn({ err, path: req.path, method: req.method }, err.message);
    }

    const body: Record<string, unknown> = {
      error: {
        code: err.code,
        message: err.message,
        // Always include details for client errors (non-5xx) so frontends can
        // read schedule lock info, format blocked reasons, etc.
        // Only strip details for 5xx in production.
        ...(err.details && (err.statusCode < 500 || !isProduction) ? { details: err.details } : {}),
        ...(!isProduction && err.stack ? { stack: err.stack } : {}),
        ...(req.requestId ? { requestId: req.requestId } : {}),
      },
    };
    res.status(err.statusCode).json(body);
    return;
  }

  // 2. ZodError
  if (err instanceof ZodError) {
    req.log.warn({ err, path: req.path, method: req.method }, 'Validation error');
    const body: Record<string, unknown> = {
      error: {
        code: 'VALIDATION_ERROR',
        message: 'Invalid parameters',
        ...(!isProduction ? { details: err.flatten() } : {}),
        ...(!isProduction && err.stack ? { stack: err.stack } : {}),
        ...(req.requestId ? { requestId: req.requestId } : {}),
      },
    };
    res.status(400).json(body);
    return;
  }

  // 3. Prisma known errors
  const prismaMapping = mapPrismaError(err);
  if (prismaMapping) {
    req.log.warn({ err, path: req.path, method: req.method }, prismaMapping.message);
    const body: Record<string, unknown> = {
      error: {
        code: prismaMapping.code,
        message: prismaMapping.message,
        ...(!isProduction && err.stack ? { stack: err.stack } : {}),
        ...(req.requestId ? { requestId: req.requestId } : {}),
      },
    };
    res.status(prismaMapping.statusCode).json(body);
    return;
  }

  // 4. Unknown errors — 500
  req.log.error({ err, path: req.path, method: req.method }, 'Unhandled error');

  captureException(err, {
    requestId: req.requestId,
    path: req.path,
    method: req.method,
    userId: req.userId,
  });

  const body: Record<string, unknown> = {
    error: {
      code: 'INTERNAL_ERROR',
      message: 'Internal server error',
      ...(!isProduction && err.stack ? { stack: err.stack } : {}),
      ...(req.requestId ? { requestId: req.requestId } : {}),
    },
  };
  res.status(500).json(body);
}
